协程

1. 协程是什么？

协程是轻量级线程，但不是线程。一个线程内可以运行成千上万个协程。

Kotlin 协程是一种轻量级的线程机制，用于优雅地处理异步任务。它通过挂起函数（suspend）和协程构建器（如 launch 和 async）实现非阻塞式的异步编程，简化了复杂的回调逻辑。

// 线程 vs 协程对比
fun compareThreadAndCoroutine() {
    // 1. 线程 - 重量级
    repeat(100_000) { i ->
        Thread { 
            Thread.sleep(1000L)
            println("Thread $i")
        }.start()  // 创建10万个线程 - 内存爆炸！
    }
    
    // 2. 协程 - 轻量级
    repeat(100_000) { i ->
        GlobalScope.launch {
            delay(1000L)  // 非阻塞挂起
            println("Coroutine $i")
        }  // 创建10万个协程 - 轻松应对
    }
}

// 三种主要构建器

    // 1. launch - 启动一个不返回结果的协程
    val job = GlobalScope.launch {
        println("launch: 后台任务")

        println("runBlocking: 阻塞测试")
    }
}

fun showDispatchers() {
    // 1. Dispatchers.Main - 主线程，更新UI
    CoroutineScope(Dispatchers.Main).launch {
        textView.text = "UI更新"  // 必须在主线程
    }
    
    // 2. Dispatchers.IO - IO操作
    CoroutineScope(Dispatchers.IO).launch {
        val data = downloadFile()  // 网络请求
        withContext(Dispatchers.Main) {
            updateUI(data)  // 切回主线程
        }
    }
    
    // 3. Dispatchers.Default - CPU密集型计算
    CoroutineScope(Dispatchers.Default).launch {
        val result = calculatePrimeNumbers(1000000)  // 复杂计算
    }
    
    // 4. Dispatchers.Unconfined - 不指定线程（不常用）
    CoroutineScope(Dispatchers.Unconfined).launch {
        // 在调用者线程启动，在恢复时可能在不同线程
    }
}

// 2. 结构化并发（推荐）
class MyActivity : AppCompatActivity() {
    // 创建与生命周期绑定的作用域
    private val scope = CoroutineScope(SupervisorJob() + Dispatchers.Main)
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // 启动协程
        scope.launch {
            val user = async { fetchUser() }
            val posts = async { fetchPosts() }
            val comments = async { fetchComments() }
            
            // 等待所有结果
            updateUI(user.await(), posts.await(), comments.await())
        }
    }
    
    override fun onDestroy() {
        super.onDestroy()
        scope.cancel()  // 取消所有协程
    }
}

class UserViewModel(
    private val userRepository: UserRepository
) : ViewModel() {
    
    private val _users = MutableLiveData<List<User>>()
    val users: LiveData<List<User>> = _users
    
    private val _loading = MutableLiveData(false)
    val loading: LiveData<Boolean> = _loading
    
    // 获取用户列表
    fun loadUsers() {
        viewModelScope.launch {
            _loading.value = true
            
            try {
                // 在IO线程执行网络请求
                val userList = withContext(Dispatchers.IO) {
                    userRepository.getUsers()
                }
                
                _users.value = userList
            } catch (e: Exception) {
                // 处理异常
                _users.value = emptyList()
            } finally {
                _loading.value = false
            }
        }
    }
    
    // 并行请求多个数据
    fun loadUserData(userId: String) {
        viewModelScope.launch {
            val userDeferred = async { userRepository.getUser(userId) }
            val postsDeferred = async { userRepository.getUserPosts(userId) }
            val friendsDeferred = async { userRepository.getUserFriends(userId) }
            
            // 等待所有并行请求完成
            val user = userDeferred.await()
            val posts = postsDeferred.await()
            val friends = friendsDeferred.await()
            
            // 合并数据
            _users.value = listOf(user.copy(posts = posts, friends = friends))
        }
    }
}

class SafeViewModel : ViewModel() {
    // 方式1：try-catch
    fun loadDataWithTryCatch() {
        viewModelScope.launch {
            try {
                val data = repository.fetchData()
                // 处理数据
            } catch (e: IOException) {
                // 处理网络异常
                showError("网络错误: ${e.message}")
            } catch (e: Exception) {
                // 处理其他异常
                showError("发生错误: ${e.message}")
            }
        }
    }
    
    // 方式2：CoroutineExceptionHandler
    fun loadDataWithExceptionHandler() {
        val exceptionHandler = CoroutineExceptionHandler { _, exception ->
            showError("协程异常: ${exception.message}")
        }
        
        viewModelScope.launch(exceptionHandler) {
            val data = repository.fetchData()
            // 处理数据
        }
    }
    
    // 方式3：使用 Result 封装
    suspend fun loadDataSafely(): Result<Data> = withContext(Dispatchers.IO) {
        return@withContext try {
            Result.success(repository.fetchData())
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
}

class CancellableViewModel : ViewModel() {
    fun loadDataWithTimeout() {
        viewModelScope.launch {
            // 1. 设置超时
            val data = withTimeoutOrNull(5000) {  // 5秒超时
                repository.fetchLargeData()
            }
            
            if (data == null) {
                // 超时处理
                showError("请求超时")
            }
        }
    }
    
    fun cancellableOperation() {
        val job = viewModelScope.launch {
            repeat(1000) { i ->
                // 检查协程是否活跃
                if (isActive) {
                    // 模拟工作
                    delay(100)
                    println("处理 $i")
                } else {
                    // 清理资源
                    cleanup()
                    return@launch
                }
            }
        }
        
        // 3秒后取消
        viewModelScope.launch {
            delay(3000)
            job.cancel("用户取消")
        }
    }
}

fun fetchUserData(continuation: Continuation): Any {
    when (continuation.label) {
        0 -> {  // 初始状态
            continuation.label = 1
            return requestToken(continuation)
        }
        1 -> {  // 获取token后
            val token = continuation.result as String
            continuation.label = 2
            return fetchUser(token, continuation)

            val user = continuation.result as User
            return user
        }
    }
}

// 完整的 MVVM + 协程 + Flow 架构
class UserViewModel(
    private val userRepository: UserRepository,
    private val savedStateHandle: SavedStateHandle
) : ViewModel() {
    
    // StateFlow 状态管理
    private val _uiState = MutableStateFlow<UserUiState>(UserUiState.Loading)
    val uiState: StateFlow<UserUiState> = _uiState
    
    // 从保存的状态恢复
    private val userId: String = savedStateHandle["userId"] ?: ""
    
    init {
        loadUser()
    }
    
    // 使用 Flow 收集数据
    private fun loadUser() {
        viewModelScope.launch {
            // 从 Room 数据库观察数据变化
            userRepository.observeUser(userId)
                .flowOn(Dispatchers.IO)
                .catch { e ->
                    _uiState.value = UserUiState.Error(e.message ?: "未知错误")
                }
                .collect { user ->
                    _uiState.value = if (user != null) {
                        UserUiState.Success(user)
                    } else {
                        UserUiState.Error("用户不存在")
                    }
                }
        }
    }
    
    // 刷新数据
    fun refreshUser() {
        viewModelScope.launch {
            _uiState.value = UserUiState.Loading
            try {
                userRepository.refreshUser(userId)
            } catch (e: Exception) {
                _uiState.value = UserUiState.Error(e.message ?: "刷新失败")
            }
        }
    }
    
    // 密封类表示状态
    sealed class UserUiState {
        object Loading : UserUiState()
        data class Success(val user: User) : UserUiState()
        data class Error(val message: String) : UserUiState()
    }
}

// Repository 层
class UserRepository @Inject constructor(
    private val userApi: UserApi,
    private val userDao: UserDao
) {
    // 网络请求
    suspend fun refreshUser(userId: String) {
        val user = userApi.getUser(userId)  // 网络请求
        userDao.insertUser(user)  // 存入数据库
    }
    
    // 观察数据库变化
    fun observeUser(userId: String): Flow<User?> {
        return userDao.observeUserById(userId)
    }
}

总结

viewModelScope(协程作用域) 的核心价值：

自动生命周期管理 - 与 ViewModel 生命周期绑定

避免内存泄漏 - ViewModel 销毁时自动取消协程

简化代码 - 不需要手动管理协程作用域

线程安全 - 默认在主线程启动

协程的关键特性：

轻量级 - 一个线程可运行大量协程

结构化并发 - 父子协程关系明确

挂起非阻塞 - 提高线程利用率

流式API - 支持 Flow 响应式编程

最佳实践原则：

始终在 ViewModel 中使用 viewModelScope启动协程

耗时操作切换到 Dispatchers.IO

UI 更新切换回 Dispatchers.Main

正确处理异常和取消

使用 Flow 进行响应式数据流处理

通过合理使用 viewModelScope和协程，可以构建出高效、可维护、无泄漏的 Android 应用架构。
